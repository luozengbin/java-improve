23 /**
24  * A simple EJB 3.0 stateless session bean with a Local business interface
25  * that uses EJB 3.0 interceptor mechanism to validate the comments entered
26  * by reviewer.
27  * @Interceptors({AuditInterceptor.class}) annotation specifies the
28  *  interceptor class that is associated with the bean file.
29  */
30 @Stateless
31 @Interceptors( { AuditInterceptor.class })
32 @Local(InterceptedProcess.class)
33 public class InterceptedProcessBean {
34 
35   //reference name and type inferred from variable.
36   @EJB ReviewListener listener;
37   @PersistenceContext(unitName = "reviewSession")
38   private EntityManager em;
39 
40   /**
41    * The method-level @ExcludeClassInterceptors annotation specifies that
42    * the business method specified in the associated interceptor class
43    * (AuditInterceptor) should not be invoked for the getRandomBook method.
44    */
45  @ExcludeClassInterceptors
46   public Book getRandomBook() {
47     List<Book> books = em.createQuery("select i from Book i").getResultList();
48     int size = books.size();
49     return size == 0 ? null : books.get(new Random().nextInt(size));
50   }
51 
52   public void addReview(int uid, String reviewerName, int rating, String comments) {
53     Book book = em.find(Book.class, uid);
54     if (book == null)
55         throw new ObjectNotFoundException("Can't find book with id ["+uid+"]");
56     Reviewer reviewer = em.find(Reviewer.class, reviewerName);
57     if (reviewer == null) {
58       reviewer = new Reviewer(reviewerName, Person.Gender.MALE);
59     }
60     Review review = reviewer.review(book, rating, comments);
61     em.persist(reviewer);
62     listener.reviewAdded(review);
63   }
64 }
