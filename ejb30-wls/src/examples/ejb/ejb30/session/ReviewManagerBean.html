17 /**
18  * A Stateless Session Bean operating on a persistent domain model.
19  * The EJB container registers the bean in JNDI global namespace.
20  * The JNDI name for the bean is the <em>simple</em> name (i.e.
21  * stripped of package name) of the business interface it implements.
22  * For this example, the JNDI name will be <code>ReviewService</code>
23  * which is specified by the annotation element mappedName="ReviewService".<p>
24  * <B>NOTE</B>:This policy is ambiguous when two beans implements the same
25  * business interface.<p>
26  * <p/>
27  * The EJB contain also injects the bean with a instance of
28  * {@link javax.persistence.EntityManager EntityManager} (or rather a proxy).
29  * This EntityManager is used to interact with the persistent EJB domain
30  * model.
31  *
32  * @author <A HREF="mailto:ppoddar@bea.com>Pinaki Poddar</A>
33  */
34 @Stateless(mappedName="ReviewService")
35 @Remote(ReviewService.class)
36 public class ReviewManagerBean extends ReviewServiceImpl
37     implements ReviewService {
38   @PersistenceContext(unitName = "reviewSession")
39   private EntityManager em;
40   @EJB ReviewListener listener;
41 
42   @PostConstruct
43   public void init(){
44      // inject the EnityManager.
45      injectEntityManager(em);
46      addReviewAddedListener(listener);
47   }
48 
49 
50 }
