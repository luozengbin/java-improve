011 /** 
012  * Item, a Pure Old Java Object (POJO), is an abstract description of a thing  
013  * that a {@link Reviewer} can review. 
014  * Item is annotated as an Entity Java Bean by <code>@Entity</code> annotation,
015  * but it does not require to extend any any framework classes or implement any 
016  * particular interface according to EJB 3.0 Specification. <BR>
017  * This class and its fields are annotated with O-R Mapping annotations
018  * to describe the database schema that would contain the persisted state of
019  * the instances. This class annotates the fields. A class can also employ
020  * property based annotations by annotatating its getter methods. 
021  * Alternatively, the O-R mapping can be configured via 
022  * deployment descriptors. The deployment descriptor specification always
023  * overrides any annotations.<BR> 
024  * Item is also the root of an inheritance hierarchy for 
025  * {@link examples.ejb.ejb30.domain.Book Book}, 
026  * {@link examples.ejb.ejb30.domain.Music Music} and
027  * {@link examples.ejb.ejb30.domain.Movie Movie}. The EJB 3.0
028  * domain model allows an abstract (or even non-persistent) class to be
029  * superclass of persistent domain classes. <BR>
030  * The inheritance strategy i.e. how the subclasses be represnted in the
031  * database table(s), are specified in this base class.<BR>
032  * <BR>
033  * Following is the class-level annotation.<BR>
034  * <pre>
035   @Entity
036   @Table(name = "ITEM")
037   @Inheritance(strategy = InheritanceType.SINGLE_TABLE)
038   @DiscriminatorColumn(name = "ITEM_TYPE",
039   discriminatorType = DiscriminatorType.STRING,length = 8)
040  * </pre>
041  *  <code>@Entity</code> marks the Item class as a persistent entity.<BR>
042  *  <code>@Table</code> specifies the relatonal database table name that 
043  *  would store the records of Item and its subclasses.<BR>
044  *  <code>@Inheritance</code> qualifies that the instances of all the classes
045  *  derived from Item would be stored in a <code>SINGLE TABLE</code><BR>
046  *  <code>@DiscriminatorColumn</code> is used to identify the database column 
047  *  that will contain values to distinguish different subclasses as all of them 
048  *  are stored in the same table. Here we use a column <code>ITEM_TYPE</code>
049  *  of variable character length of 8.<BR>
050  *  
051  *  This class also uses annotation to descibe a simple named query in 
052  *  Java Persistence Query Language (JPQL).<BR>
053  *  <pre>
054  *  @NamedQuery(name = "Item.findByTitle",
055  *  query = "SELECT i FROM Item i WHERE i.title LIKE :title")
056  *  </pre>
057  * This named query finds Items whose name matches with the a given title. 
058  * The parameter <code>:title</code> will be bound to this query at runtime.
059  * The query names are globally scoped and hence we follow a naming convention
060  * to avoid possible name clash by appending the class name.<code> 
061  *
062  */
063 @NamedQuery(name = "Item.findByTitle", 
064     query = "SELECT i FROM Item i WHERE i.title LIKE :title")
065 @Entity
066 @Table(name = "ITEM")
067 @Inheritance(strategy = InheritanceType.SINGLE_TABLE)
068 @DiscriminatorColumn(name = "ITEM_TYPE", discriminatorType = DiscriminatorType.STRING, length = 8)
069 public class Item
070   implements Serializable
071 {
072   /** 
073    * Unique identifier for each item. 
074    * The field is annotated as
075    * <pre>
076    @Id
077    @GeneratedValue
078    * </pre>
079    * <code>@Id</code> specifies that the field value would uniquely identify an
080    * instance.
081    * <code>@GeneratedValue</code> designates that the Persistence Runtime Provider
082    * would generate unique values.
083    */

090    * The field is annotated with <code>@Column</code> and would be stored in

097   /** Average rating of this item. 
098    *  This field is annotated as <code>@Transient</code> meaning that the value
099    *  of this field is not stored in the database. The rating of an Item
100    *  is computed by averaging ratings of all the reviews for this Item. This is
101    *  an example of transient field whose value depends on other field values.
102    * 
103    * @see #calculateRating() on how and <em>when</em> to compute this field value. 
104    */
105   @Transient
106   private float rating;
107 
108   /** 
109    * Refers to the {@link Artist Artist} who created this item. Based on concrete
110    * Item type, this field would imply a Author of Book, Singer for Music or 
111    * Director for Movie.<BR>
112    * An Item refers to one and only one Artist. Many Items can refer to the same
113    * Artist. Artist <em>does not</em> refer to the Item. Hence, this is an 
114    * example to unidirectional single-valued relationship.</BR>
115    * This relationship is annotated as 
116    * <pre>
117    @ManyToOne(cascade = CascadeType.PERSIST)
118    * </pre>
119    * The unidirectional single-valued relationship can be specified as either
120    * a unidirectional <code>@OneToOne</code> or as a unidirectional 
121    * <code>@ManyToOne</code> annotation. As there can be more than one item 
122    * created by the same Artist, we use a <code>@ManyToOne</code> annotation.
123    * <BR>
124    * The <code>cascade</code> qualifies that the linked Artist will be persisted
125    * when an Item is persisted.   
126    * 
127    * 
128    */
129   @ManyToOne(cascade = CascadeType.PERSIST)
130   @Column(name = "ARTIST")
131   private Artist artist;

133   /** 
134    * Refers to all the {@link Review reviews} on this item.
135    * An Item can have many {@link Review Reviewes}. Each Review refers to its
136    * Item. This is an example of bidirectional, one-to-many relationship.<BR>
137    * This relationship is annotated as 
138    * <pre>
139    @OneToMany(cascade = CascadeType.ALL, mappedBy = "reviewed")
140    * </pre> 
141    * Item owns the relationship in the sense if an item is deleted so all its
142    * reviewes are deleted as well (but not vice versa).<BR>
143    * The annotation specifies the name of the field in the referred class 
144    * {@link Review Review} that holds the other terminal of this particular 
145    * relationship.
146    * Also a <code>@OrderBy("createdDate DESC")</code> annotation is used to sort
147    * the reviews by their date of creation. 
148    */
149   @OneToMany(cascade = CascadeType.ALL, mappedBy = "reviewed")
150   @OrderBy("createdDate DESC")
151   private List<Review> reviews;

166   @ManyToMany(mappedBy = "watchedItems")
167   private List<Reviewer> watchers;

288   /** 
289    * Adds a review to this item. Updates the average rating on the item by 
290    * the rating of the new review.<BR>
291    * This is an example of a <em>business</em>
292    * method that modifies properties of the domainn instance.
293    * The EJB3.0 specification recommends <em>not<em> to use direct access 
294    * to field values within a <em>business</code> method but to access them 
295    * via the accessors. 
296    * 
297    * @param r review that would be added. 
298    * @return true if the given review has been added to this item. false
299    * otherwise. 
300    */
301   public boolean addReview (Review r)
302   {
303     if (r == null)
304       return false;
305     if (!this.equals (r.getReviewed ()))
306       return false;
307 
308     if (getReviews () == null)
309       setReviews (new ArrayList<Review> ());
310 
311     int ratedBy = getReviews ().size ();
312     rating = (rating * ratedBy + r.getRating ()) / ratedBy + 1;
313 
314     getReviews ().add (0, r);
315     return true;
316   }

330   /**
331    * Calculates the rating for this receiver by averaging the rating for each
332    * associated Reviews. <BR> 
333    * Callback methods in entity classes are called in different points in
334    * life cycle of a managed entity. This method is annotated with one such
335    * <code>@PostLoad</code> callback. This method will be invoked by the 
336    * Persistence Provider Runtime <em>after</em> the instance state is loaded 
337    * from the database.<BR>
338    * In this method, we calculate one property: the average
339    * rating. This is a transient property and is calculated as the
340    * average the ratings over all the reviews on this item.<BR>   
341    * This average can be calculated only when all the review ratings are 
342    * loaded. When value of a property depends on the values
343    * of some other properties, it should only be calculated when other
344    * properties have been loaded. As Persistence Provider Runtime can
345    * load the persistent properties at an unspecified order, these callback
346    * methods are the best place to warranty that the persistent properties
347    * of the instance have been loaded.  
348    * 
349    *
350    */
351   @PostLoad
352   public void calculateRating ()
353   {
354     if (getReviews () == null || getReviews ().isEmpty ())
355     {
356       return;
357     }
358     float totalRating = 0f;
359     for (Review r : getReviews ())
360       totalRating += r.getRating ();
361     rating = totalRating / getReviews ().size ();
362   }
