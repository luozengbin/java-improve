06 import javax.persistence.*;

08 /** 
09  *  Music is a kind of Item. 
10  *  The target table for Music is decided by the inheritance strategey 
11  *  specified in its {@link examples.ejb.ejb30.domain.Item superclass}. In this 
12  *  case, <code>SINGLE_TABLE</code>
13  *  inheritance is used -- that stores all subclasses of Item including
14  *  Music to a single table. <BR>
15  *  Hence it is required to discriminate the records of Music from records
16  *  of other types. The <code>@DiscriminatorValue</code> annotation specifies 
17  *  that
18  *  all records of Music will carry the value <code>"MUSIC"</code> in the
19  *  <code>@DiscriminatorColumn</code> specified with the 
20  *  {@link Item inheritance strategy}. 
21  *   
22  * @author <A HREF="mailto:ppoddar@bea.com>Pinaki Poddar</A>
23  *
24  */
25 
26 @Entity
27 @DiscriminatorValue(value = "MUSIC")
28 public class Music
29   extends Item
30 {
31 
32   /* 
33    * An additional field to specify duration of this music in minutes.
34    * The same column name <code>DURATION</code> is used for the field
35    * of sibling class {@link Movie Movie}. Check the database 
36    * schema to see how these two properties are represnted in the table. 
37    * 
38    */
39   @Column(name = "DURATION")
40   private Integer duration;
41 
42   /** 
43    * A no-arg constructor is required for enhancement.
44    * 
45    *
46    */
47   public Music ()
48   {
49     super ();
50   }
51 
52   /** 
53    * The public constructor constructs with a title.
54    * 
55    * @param title
56    */
57   public Music (String title)
58   {
59     super (title);
60   }
61 
62   public Integer getDuration ()
63   {
64     return duration;
65   }
66 
67   public void setDuration (int d)
68   {
69     if (d < 0)
70       throw new IllegalArgumentException ("Invalid duration " + d
71           + " for " + this);
72 
73     duration = d;
74   }
