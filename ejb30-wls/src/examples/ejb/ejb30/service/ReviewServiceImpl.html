021 /**
022  * A service class that operates on the {@link examples.ejb.ejb30.domain domain
023  * model}. A persistent domain model can be directly worked on with the
024  * {@link javax.persistence.EntityManager  EntityManager} interface. However,
025  * this receiver provides a stateless service_ to operate on the persistent
026  * domain objects. <BR>
027  * This service_ closely resembles what a Stateless Session Bean that operates on
028  * the persistent domain model.
029  *
030  * @author <A HREF="mailto:ppoddar@bea.com>Pinaki Poddar</A>
031  */

131   public Review newReview(Reviewer reviewer, Item item, int rating,
132                           String comment) {
133     EntityManager em = getEntityManager();
134     beginTransaction();
135     item = em.merge(item);
136     reviewer = em.merge(reviewer);
137     Review review = reviewer.review(item, rating, comment);
138     em.persist(review);
139     commit();
140     for (ReviewListener listener : listerners) {
141        listener.reviewAdded(review);
142     }
143     return review;
144   }

192     em.persist(item);
193     commit();
194

199   public boolean subscribe(String reviewerName, int itemId) {

201     beginTransaction();
202     Item item = em.find(Item.class, itemId);
203     if (item == null)
204       throw new ObjectNotFoundException("No item with id " + itemId);
205     Reviewer reviewer = em.find(Reviewer.class, reviewerName);

248   public boolean removeFeededReview(String reviewerid, int reviewid) {
249     EntityManager em = getEntityManager();
250     beginTransaction();
251     Reviewer reviewer = em.find(Reviewer.class, reviewerid);
252     if (reviewer == null)
253       throw new ObjectNotFoundException("reviewer" + reviewerid + " is not registered");
254     List<Review> watches = reviewer.getWatchedReviews();
255     Review toBeRemoved = null;
256     for (Review review : watches) {
257       if (review.getId() == reviewid) {
258         toBeRemoved = review;
259         break;
260       }
261     }
262     watches.remove(toBeRemoved);
263     commit();
264     return toBeRemoved == null;
265   }
