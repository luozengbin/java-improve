20   /**
21    * a business method interceptor method.
22    */
23   @AroundInvoke
24   public Object audit(InvocationContext ic) throws Exception {
25     Method method = ic.getMethod();
26     if (method.getName().equals("addReview")) {
27       Object[] parameters = ic.getParameters();
28       String comments = (String) parameters[parameters.length - 1];
29       if (comments == null || comments.trim().length() == 0) {
30         throw new BadCommentsException("empty comments is not allowed, please enter your comments");
31       }
32       if (comments.indexOf("bitch") > -1) {
33         throw new BadCommentsException("the comments entered includes impolite word, please remove it.");
34       }
35     }
36     return ic.proceed();
37   }
38 
39   /**
40    * lifecycle callback interceptor method.
41    */
42   @PreDestroy
43   public void preDestroy(InvocationContext ic) throws Exception {
44     System.out.println("Invoking method: " + ic.getMethod());
45   }
