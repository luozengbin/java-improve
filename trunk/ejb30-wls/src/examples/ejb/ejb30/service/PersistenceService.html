04 package examples.ejb.ejb30.service;

06 import javax.persistence.EntityManager;
07 import javax.persistence.EntityManagerFactory;
08 import javax.persistence.Persistence;
09 
10 /**
11  * PersistenceService provides basic support for a transactional application
12  * using a {@link javax.persistence.EntityManager EntityManager}. <BR>
13  * The service_ can be initialized/constructed in following ways<BR>
14  * <LI>providing a configuration file that specifies the concrete class name of
15  * the Persistence Provider Runtime implementation of
16  * {@link javax.persistence.EntityManagerFactory EntityManagerFactory}.
17  * <LI>supplying a {@link javax.persistence.EntityManagerFactory
18  * EntityManagerFactory}. <BR>
19  * This service uses a session-per-thread model to provide the caller with an
20  * EntityManager when the EntityManager is not injected by calling injectEntityManager().
21  * This is done by attaching a EntityManager to the calling
22  * thread.
23  *
24  * @author <A HREF="mailto:ppoddar@bea.com>Pinaki Poddar</A>
25  */
26 public abstract class PersistenceService {
27   private EntityManagerFactory _emf;
28   private EntityManager _em;
29 
30   private ThreadLocal<EntityManager> _thread = new ThreadLocal<EntityManager>();
31 
32   private boolean _isManaged;
33 
34   public PersistenceService() {
35   }
36 
37   public PersistenceService(EntityManagerFactory emf) {
38     if (emf == null)
39       throw new NullPointerException("null-emf");
40     _emf = emf;
41   }
42 
43   /**
44    * Gets a entity manager. The entity manager is one per-calling thread.
45    *
46    * @return
47    */
48   public EntityManager getEntityManager() {
49     EntityManager em = getEM();
50     if (em == null)
51       throw new NullPointerException("no-entity-manager");
52     return em;
53   }
54 
55   public void injectEntityManager(EntityManager em) {
56     _isManaged = true;
57     _em = em;
58   }
59 
60   private final EntityManager getEM() {
61     if (_em != null)
62       return _em;
63     if (_thread == null)
64       _thread = new ThreadLocal<EntityManager>();
65     EntityManager em = _thread.get();
66     if (em != null)
67       return em;
68     if (_emf == null)
69       return null;
70     em = _emf.createEntityManager();
71     _thread.set(em);
72     return em;
73   }
74 
75   protected final EntityManager beginTransaction() {
76     EntityManager em = getEntityManager();
77     if (!_isManaged && !em.getTransaction().isActive())
78       em.getTransaction().begin();
79     return em;
80   }
81 
82   protected final void commit() {
83     if (_isManaged)
84       return;
85     EntityManager em = getEntityManager();
86     if (em.getTransaction().isActive())
87       em.getTransaction().commit();
88   }
89 
90   protected final void rollback() {
91     if (_isManaged)
92       return;
93     EntityManager em = getEntityManager();
94     if (em.getTransaction().isActive())
95       em.getTransaction().rollback();
96   }
97 
98 }
