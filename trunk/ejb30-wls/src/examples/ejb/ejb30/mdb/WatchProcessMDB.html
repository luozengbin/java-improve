06 import javax.annotation.PostConstruct;
07 import javax.annotation.PreDestroy;
08 import javax.ejb.ActivationConfigProperty;
09 import javax.ejb.MessageDriven;
10 import javax.ejb.TransactionAttribute;
11 import javax.jms.JMSException;
12 import javax.jms.Message;
13 import javax.jms.MessageListener;
14 import javax.persistence.EntityManager;
15 import javax.persistence.PersistenceContext;
16 
17 /**
18  *
19  * MDB asynchronously feeds the new added reviews to subscribers.
20  *
21  */
22 
23 @MessageDriven
24     (mappedName = "weblogic.examples.ejb30.ExampleQueue",
25         name = "WatchProcessMDB",
26         activationConfig = {
27         @ActivationConfigProperty(propertyName = "destinationType", propertyValue = "javax.jms.Queue")
28             }
29     )
30 public class WatchProcessMDB implements MessageListener {
31   @PersistenceContext(unitName = "reviewSession")
32   private EntityManager em;
33   private SubscriptionService service;
34 
35   @PostConstruct
36   private void init(){
37     service = new SubscriptionService();
38     // inject the EntityManager;
39     service.injectEntityManager(em);
40   }
41   // 
42   @TransactionAttribute(value=javax.ejb.TransactionAttributeType.REQUIRED)
43   public void onMessage(Message msg) {
44     int reviewid = -1;
45     try {
46       reviewid = msg.getIntProperty("reviewid");
47     } catch (JMSException e) {
48       throw new RuntimeException(e);
49     }
50     service.pushWatchedReview(reviewid);
51   }
52 
53 }
