06 import examples.ejb.ejb30.domain.Book;
07 import examples.ejb.ejb30.domain.Artist;
08 
09 import javax.ejb.Local;
10 import javax.ejb.LocalHome;
11 import javax.ejb.Stateless;
12 import javax.persistence.PersistenceContext;
13 import javax.persistence.EntityManager;
14 
15 /**
16  * This ItemManagerBean session bean is annotated with a business
17  *  interface and Home interface, thereafore it can be invoked by
18  * client code following the EJB2.1 spec.
19  */
20 @Stateless
21 @Local(ItemManager.class)
22 @LocalHome(ItemManagerHome.class)
23 public class ItemManagerBean{
24   @PersistenceContext(unitName = "reviewSession")
25   private EntityManager em;
26 
27   public Book addBook(String title, String creatator) {
28     Artist artist = em.find(Artist.class, creatator);
29     if (artist == null) {
30       artist = new Artist(creatator);
31       em.persist(artist);
32     }
33     Book book = new Book(title);
34     book.setArtist(artist);
35     em.persist(book);
36     return book;
37   }
38 }
