016 /**
017  * The business interface for Stateful Session Bean ReviewStatManagerBean.
018  */
019 public interface ReviewStatManager{
020 
021   /**
022    * Gets the reviews added.
023    * @return
024    */
025   public int getAddedReviewCount();
026 
027   /**
028    * Gets the items added.
029    * @return the number of items has been added.
030    */
031   public int getAddedItemCount();
032 
033   /**
034    *
035    * @param username  the name of the reviwer.
036    * @param password  the password of the reviewer.
037    * @return  return true when the a reviewer with the specified username and password.
038    */
039   public boolean login(String username, String password);
040 
041   /**
042    * log out.
043    */
044   public void logOut();
045 
046   /**
047    * Add a review on a specified Item.
048    * @param itemId  the unique identifier of the item.
049    * @param rating  the rating.
050    * @param comment
051    * @return  a newly-created and detached review.
052    * @throws ObjectNotFoundException
053    */
054   public Review newReview(int itemId, int rating, String comment) throws ObjectNotFoundException;
055 
056   /**
057    * Create and persitent a new Item.
058    * @param itemType  the type of item. valid option are "Book","Music","Movie"
059    * @param title the title of the item.
060    * @param artistName  the name of the creator.
061    * @return  a newly-created and detached Item.
062    * @throws ObjectExistedException
063    */
064   public Item newItem(String itemType, String title, String artistName) throws ObjectExistedException;
065 
066   /**
067    * Return a list of reviews for items which are watched the logged in user.
068    * @return
069    * @throws NotLoggedInException
070    */
071   public List<Review> getFeededReviews() throws NotLoggedInException;
072 
073   /**
074    * Remove a certain review from the watched list.
075    * @param reviewid
076    * @throws NotLoggedInException
077    * @throws ObjectNotFoundException
078    */
079   public void removeFeededReview(int reviewid) throws NotLoggedInException,ObjectNotFoundException;
080 
081   /**
082    * whether user has been logged in.
083    * @return
084    */
085   public boolean isLogged();
086 
087   /**
088    *  subscribe to new reviews on item with the given itemId.
089    * @param itemId
090    * @return
091    * @throws NotLoggedInException user has to login before subscribe to new reviews on certain item.
092    * @throws ObjectNotFoundException  Item with the given itemId is not existed.
093    */
094   public boolean subscribe(int itemId) throws NotLoggedInException,ObjectNotFoundException;
095 
096   /**
097    *  subscribe to new reviews on item with the given itemId.
098    * @param itemId
099    * @return
100    * @throws NotLoggedInException user has to login before subscribe to new reviews on certain item.
101    * @throws ObjectNotFoundException  Item with the given itemId is not existed.
102    */
103   public boolean unSubscribe(int itemId) throws NotLoggedInException,ObjectNotFoundException;

105 }
