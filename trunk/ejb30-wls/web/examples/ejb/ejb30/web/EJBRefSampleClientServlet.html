031    * @param request

033    */
034   private void printPageHeader(HttpServletRequest request, HttpServletResponse response, PrintWriter out) {
035     try {
036       request.getRequestDispatcher("/header.html").include(request, response);
037     } catch (Exception e) {
038       printErrorMsgs(out, e);
039       return;
040     }
041     out.println("<link href=\"sample.css\" rel=\"stylesheet\"></link>");
042     String tips = "This example demostrates the usage of EJB reference in Session Bean and Servlet.<p>" +
043         "The <a href='" + Constants.JAVA_SOURCE_BASE_DIR + "/web/EJBRefSampleClientServlet.html'>EJBRefSampleClientServlet</a> " +
044         "references a simple stateless session bean <a href='" + Constants.JAVA_SOURCE_BASE_DIR + "/ejbref/ProcessBean.html'>ProcessBean</a>" +
045         " which further references another session bean ReviewerListenerBean <a href='" + Constants.JAVA_SOURCE_BASE_DIR + "/mdb/ReviewerListenerBean.html'><a/><p>" +
046         "The ProcessBean will select a book for the guest reviewer to add comments randomly. " +
047         "When new comments are added, ReviewerListenerBean is called by the ProcessBean to " +
048         "notify the subscribers of the selected book." +
049         "<HR>";
050     out.println(tips);
051   }

128   /**
129    * Post method
130    *
131    * @param request
132    * @param response
133    * @throws java.io.IOException
134    * @throws javax.servlet.ServletException
135    */
136   public void doPost(HttpServletRequest request, HttpServletResponse response)
137       throws IOException, ServletException {
138     response.setContentType("text/html");
139     String errorMsg = null;
140     int rate = 0;
141     int uid = 0;
142 
143     String bookid = request.getParameter("selectedID");
144     if (bookid == null || bookid.trim().equals("")) {
145       errorMsg = "Please select the book you want to review";
146     } else {
147       uid = Integer.parseInt(bookid);
148     }
149 
150     String rating = request.getParameter("rating");
151     String comments = request.getParameter("comments");
152 
153     //doGet(request, response);
154     PrintWriter out = response.getWriter();
155     response.setContentType("text/html");
156     printPageHeader(request, response, out);
157 
158     try {
159       rate = Integer.parseInt(rating);
160       if (rate < 0 || rate > 5) {
161         errorMsg = "Please input rating as a number between 0 and 5";
162       }
163     } catch (Exception e) {
164       errorMsg = "Please input rating as a number";
165     }
166 
167     // there is no error
168     if (errorMsg == null) {
169       service_.addReview(uid, "guest", rate, comments);
170       out.println("<p align='center'>Your comments has been saved. Please review another randomly selected book.</p>");
171     } else
172       out.println("<p align='center'><font color='#FF0000'>" + errorMsg + "</font></p>");
173 
174     printBookTable(out);
175   }
